const fs = require('fs');
const path = require('path');
 
// Expanded mock directory structure
const MOCK_DIRS = {
  root: 'src/mocks',
  googleApps: 'src/mocks/googleApps',
  services: 'src/mocks/services',
  helpers: 'src/mocks/helpers',
  fixtures: 'src/mocks/fixtures',  // Added for test data
  errors: 'src/mocks/errors'       // Added for error scenarios
};
 
// All Google Apps mocks from jest.setup.js
const GOOGLE_APPS_MOCKS = {
  'spreadsheetApp.js': 'SpreadsheetApp',
  'htmlService.js': 'HtmlService',
  'urlFetchApp.js': 'UrlFetchApp',
  'logger.js': 'Logger',
  'session.js': 'Session',
  'properties.js': 'PropertiesService'
};
 
// All service mocks from test files
const SERVICE_MOCKS = {
  'sheetHandler.js': 'src/tests/handlers/sheetHandler.test.js',
  'menuHandler.js': 'src/tests/ui/menuHandler.test.js',
  'imageHandler.js': 'src/tests/handlers/imageHandler.test.js'
};
 
// Test data and fixtures
const FIXTURE_MOCKS = {
  'testData.js': {
    source: 'src/tests/handlers/sheetHandler.test.js',
    exportName: 'TEST_DATA'
  },
  'errorScenarios.js': {
    source: 'src/tests/handlers/sheetHandler.test.js',
    exportName: 'ERROR_SCENARIOS'
  },
  'edgeCases.js': {
    source: 'src/tests/handlers/sheetHandler.test.js',
    exportName: 'EDGE_CASES'
  }
};
 
// Create all mock directories
function createMockDirectories() {
  Object.values(MOCK_DIRS).forEach(dir => {
    if (!fs.existsSync(dir)) {
      fs.mkdirSync(dir, { recursive: true });
      console.log(`✅ Created mock directory: ${dir}`);
    }
  });
}
 
// Extract all Google Apps mocks
function extractGoogleAppMocks() {
  const setupContent = fs.readFileSync('jest.setup.js', 'utf8');
  
  Object.entries(GOOGLE_APPS_MOCKS).forEach(([fileName, serviceName]) => {
    const mockContent = `// Generated from jest.setup.js
const ${serviceName} = ${setupContent.match(new RegExp(`global.${serviceName} = ({[^;]+})`))[1]};
 
module.exports = ${serviceName};`;
 
    fs.writeFileSync(
      path.join(MOCK_DIRS.googleApps, fileName),
      mockContent
    );
    console.log(`✅ Created Google Apps mock: ${fileName}`);
  });
}
 
// Extract all service mocks
function extractServiceMocks() {
  Object.entries(SERVICE_MOCKS).forEach(([fileName, sourcePath]) => {
    if (fs.existsSync(sourcePath)) {
      const testContent = fs.readFileSync(sourcePath, 'utf8');
      const mockSetup = testContent.match(/mockSheet = ({[^;]+});/)[1];
 
      const mockContent = `// Generated from ${sourcePath}
const mockService = ${mockSetup};
 
module.exports = mockService;`;
 
      fs.writeFileSync(
        path.join(MOCK_DIRS.services, fileName),
        mockContent
      );
      console.log(`✅ Created service mock: ${fileName}`);
    }
  });
}
 
// Extract all test fixtures
function extractFixtures() {
  Object.entries(FIXTURE_MOCKS).forEach(([fileName, config]) => {
    if (fs.existsSync(config.source)) {
      const testContent = fs.readFileSync(config.source, 'utf8');
      const fixtureData = testContent.match(
        new RegExp(`const ${config.exportName} = ({[^;]+});`)
      )[1];
 
      const fixtureContent = `// Generated from ${config.source}
const ${config.exportName} = ${fixtureData};
 
module.exports = ${config.exportName};`;
 
      fs.writeFileSync(
        path.join(MOCK_DIRS.fixtures, fileName),
        fixtureContent
      );
      console.log(`✅ Created fixture: ${fileName}`);
    }
  });
}
 
// Create helper functions file
function createHelperFunctions() {
  const helperContent = `// Test helper functions from jest.setup.js
const testHelpers = ${JSON.stringify(global.testHelpers, null, 2)};
 
module.exports = testHelpers;`;
 
  fs.writeFileSync(
    path.join(MOCK_DIRS.helpers, 'testHelpers.js'),
    helperContent
  );
  console.log('✅ Created test helpers file');
}
 
// Main function
async function createMockStructure() {
  console.log('🔄 Creating complete mock directory structure...');
  
  createMockDirectories();
  extractGoogleAppMocks();
  extractServiceMocks();
  extractFixtures();
  createHelperFunctions();
  
  console.log('\n✨ Mock structure creation complete!');
}
 
// Run script
createMockStructure().catch(console.error);
